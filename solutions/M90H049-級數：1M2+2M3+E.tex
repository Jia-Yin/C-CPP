\section{M90H049 - 級數 $1\times 2+2\times 3+3\times 4+\cdots$}
考慮數列 $1\times 2+2\times 3+3\times 4+\cdots$。 
輸入一整數n，輸出該數列前n項之和。
\subsection{解題思惟}
\begin{enumerate}
	\item 初始化變數ans=0，用來作為累加之用。
	\item 跑for迴圈令i從1到n，每次對ans累加$i\times(i+1)$。
\end{enumerate}
\subsection{程式碼}
\begin{cppcode}
	#include <stdio.h>
	
	int main()
	{
		int n, i, ans=0;
		scanf("%d", &n);
		for (i=1; i<=n; i++) {
			ans += i*(i+1);
		}
		printf("%d", ans);
		return 0;
	}
	
\end{cppcode}
