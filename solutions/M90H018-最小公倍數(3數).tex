\section{M90H018 - 最小公倍數(3數)}
輸入三整數，輸出其最小正公倍數。
\subsection{解題思惟}
\begin{enumerate}
	\item 求解最小公倍數有多種解法，最基本的是用定義，從任一個數的絕對值開始往上數，數到
	第一個可以整除三個輸入的數就是答案。
	\item 或者也可以先求兩數的最小公倍數，接著再和第三個數再求一次最小公倍數，這樣也可以得到解答。
\end{enumerate}

\subsection{程式碼}
\begin{cppcode}
#include <iostream>

using namespace std;

int main()
{
	int a, b, c, n;
	cin >> a >> b >> c;
	if (a<0) a=-a;
	for (n=a; ; n++) {
		if (n%a==0 && n%b==0 && n%c==0) break;
	}
	cout << n;
	return 0;
}
\end{cppcode}
