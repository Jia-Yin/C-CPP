\section{M90H014 - 正因數倒數和}
輸入一整數n，輸出其正因數倒數和。
\subsection{解題思惟}
\begin{enumerate}
	\item 令初始answer=0。
	\item 當輸入n除以迴圈內i整除，則令answer加上1.0/i(轉成浮點數型態)。
\end{enumerate}
\subsection{程式碼}
\begin{cppcode}
#include <iostream>

using namespace std;

int main()
{
	int n;
	cin >> n;
	float answer=0;
	
	for (int i = 1 ; i <= n ;i++) {
		if(n % i == 0) answer += 1.0/i;
	}
	cout << answer;
	return 0;
}
\end{cppcode}
