\section{M90H059 數列 $a(1)=3,\ 5a(n+1)=4a(n)+1$}
考慮數列: $a(1)=3,\ 5a(n+1)=4a(n)+1$。\\ 
輸入一整數n，輸出a(n)。
\subsection{解題思維}

\begin{enumerate}
	\item
	我們先寫 a 函式，使用遞迴方式，按題目所給的條件，終止條件是$a(1)=3$，其他則是$a(n)=(4a(n-1)+1)/5$。照著條件我們便可以完成函式
	\begin{inside}
		double a(int n)
		{
			if(n==1) return 3;
			return (4*a(n-1)+1)/5.0;
		}			
	\end{inside}
	\item 
	完成後代入主程式。
\end{enumerate} 

\subsection{程式碼}
\begin{cppcode}
	#include <iostream>

	using namespace std;
	
	double a(int n);
	
	int main()
	{
		double n;
		cin >> n;
		cout << endl << "a(" << n << ")=" << a(n);
		return 0;
	}
	
	double a(int n) {
		if(n==1) return 3;
		return (4*a(n-1)+1)/5.0;
	}
\end{cppcode}
