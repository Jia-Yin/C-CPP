\section{A030 百數反印}
輸入100個正整數，反向印出此100個數。

\subsection{解題思維}
將100個數依序讀入陣列，再逆序印出此100個數。

\subsection{程式碼}
\begin{cppcode}
	#include <stdio.h>
	
	int main()
	{
		int a[100];
		
		for (int i=0; i<100; i++) scanf("%d", &a[i]);
		for (int i=99; i>=0; i--) printf(" %d", a[i]);
		
		return 0;
	}
\end{cppcode}
