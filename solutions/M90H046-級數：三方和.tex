\section{M90H046 - 級數：三方和}
輸入一整數n，輸出$1^3+2^3+3^3+\cdots+n^3$。
\subsection{解題思惟}
\begin{enumerate}
	\item 初始化變數ans=0，用來作為累加之用。
	\item 跑for迴圈令i從1到n，每次對ans累加$(i\times i\times i)$。
\end{enumerate}
\subsection{程式碼}
\begin{cppcode}
	#include <stdio.h>
	
	int main()
	{
		int n, i, ans=0;
		scanf("%d", &n);
		for (i=1; i<=n; i++) {
			ans += (i*i*i);
		}
		printf("%d", ans);
		return 0;
	}
	
\end{cppcode}
