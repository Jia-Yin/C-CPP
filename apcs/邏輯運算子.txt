#include <iostream>

using namespace std;

int main()
{
    int a, b, c, flag=1;
    cin >> a >> b >> c;
    if ((a && b) == (c != 0)) { cout << "AND" << endl; flag = 0; }
    if ((a || b) == (c != 0)) { cout << "OR" << endl; flag = 0; }
    if (((a && !b) || (!a && b)) == (c != 0)) { cout << "XOR" << endl; flag = 0; }
    if (flag) cout << "IMPOSSIBLE" << endl;
    return 0;
}
