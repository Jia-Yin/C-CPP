
#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
    int n, score[200], t, tmax, tmin;
    
    while (cin >> n) {
        tmax = -999;
        tmin = 999;
        for (int i=0; i<n; i++) {
            cin >> score[i];
            if (score[i]<60 && score[i]>tmax) tmax = score[i];
            if (score[i]>59 && score[i]<tmin) tmin = score[i];
        }
        sort(score, score+n);
        for (int i=0; i<n; i++) {
            if (i) cout << " ";
            cout << score[i];
        }
        cout << endl;
        if (tmax==-999) cout << "best case\n";
        else cout << tmax << endl;
        if (tmin==999) cout << "worst case\n";
        else cout << tmin << endl;
    }
	return 0;
}
