
#include <stdio.h>

int main()
{
    int gf[50005], n, mark[50005]={0}, group=0;
    scanf("%d", &n);
    for (int i=0; i<n; i++) scanf("%d", gf+i);
	for (int i=0; i<n; i++) {
        if (mark[i]) continue;
        group++;
        do { mark[i]=1; i=gf[i]; } while (mark[i]==0);
    }
    printf("%d\n", group);
    return 0;
}
