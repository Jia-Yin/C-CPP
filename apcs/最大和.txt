#include <iostream>

using namespace std;

int main()
{
    int n, m, a[100], sum=0, k=0;
    cin >> n >> m;
    for (int i=0; i<n; i++) {
        int rmax = 0, num;
        for (int j=0; j<m; j++) {
            cin >> num;
            if (num > rmax) rmax = num;
        }
        a[i] = rmax;
        sum += rmax;
    }
    cout << sum << endl;
    for (int i=0; i<n; i++) {
        if (sum % a[i] == 0) {
            if (k==0) k=1;
            else cout << " ";
            cout << a[i];
        }
    }
    if (k==0) cout << -1;
    return 0;
}
